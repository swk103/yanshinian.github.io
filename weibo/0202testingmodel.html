代码实做

运行时提取类属性

///  提取模型类信息字典
///
///  :param: cls 模型类
///
///  :returns: 模型类信息字典
func modelInfo(cls: AnyClass) -> Dictionary<String, String> {

    var count:UInt32 = 0
    let properties = class_copyPropertyList(cls, &count)

    println("\(cls) 属性数值 \(count) \(properties)")

    free(properties)

    return Dictionary<String, String>()
}

注意

  • class_copyIvarList 和 class_copyPropertyList都是运行时的函数，一个能够提取成员变量，一个提取属性
  • 在swift中如果使用class_copyPropertyList将无法提取到Int? 这一类的属性信息
  • 但是在 swift 中无法通过 ivar_getTypeEncoding 获取到对象信息
  • 在 swift 中同样需要 free 释放 c 语言函数创建的对象，这一点与 OC 是一致的

遍历成员变量信息

    for i in 0..<count {
        let pty = properties[Int(i)]

        let cname = property_getName(pty)
        var name = String.fromCString(cname)!

        let ctype = property_getAttributes(pty)
        let type = String.fromCString(ctype)!

        println(name + "---" + type)
    }

编写函数取出自定义类名

///  取出自定义类名
///
///  :param: type 属性类型
///
///  :returns: 自定义类名
private func customClsName(type: String) -> String {
    var tname = type.componentsSeparatedByCharactersInSet(NSCharacterSet(charactersInString: ","))[0]
    var components = tname.componentsSeparatedByCharactersInSet(NSCharacterSet(charactersInString: "\""))

    if components.count > 1 {
        let name = components[1]
        if !name.hasPrefix("NS") {
            return name
        }
    }
    return ""
}

修改提取属性类信息函数

///  提取模型类信息字典
///
///  :param: cls 模型类
///
///  :returns: 模型类信息字典
func modelInfo(cls: AnyClass) -> Dictionary<String, String> {

    var count:UInt32 = 0
    let properties = class_copyPropertyList(cls, &count)

    var infoDict = Dictionary<String, String>()
    for i in 0..<count {
        let pty = properties[Int(i)]

        let cname = property_getName(pty)
        var name = String.fromCString(cname)!

        let ctype = property_getAttributes(pty)
        var type = String.fromCString(ctype)!
        type = customClsName(type)

        infoDict[name] = type
    }
    free(properties)

    return infoDict
}
