解决办法，查找所有父类属性

定义函数 fullModelInfo，首先实现代码遍历类的继承关系

///  提取完整模型信息
///
///  :param: cls 模型类
///
///  :returns: 模型类信息字典
func fullModelInfo(cls: AnyClass) -> Dictionary<String, String> {

    var currentCls: AnyClass = cls
    while let parent: AnyClass = currentCls.superclass() {
        println(parent)

        currentCls = parent
    }

    return Dictionary<String, String>()
}

根据继承关系，依次获取类的完整信息

编写字典泛型扩展，简化字典合并操作

extension Dictionary {
    mutating func merge<K, V>(dict: [K: V]) {
        for (k, v) in dict {
            self.updateValue(v as! Value, forKey: k as! Key)
        }
    }
}

代码实现

///  提取完整模型信息
///
///  :param: cls 模型类
///
///  :returns: 模型类信息字典
func fullModelInfo(cls: AnyClass) -> Dictionary<String, String> {

    var currentCls: AnyClass = cls
    var infoDict = Dictionary<String, String>()
    while let parent: AnyClass = currentCls.superclass() {
        // 追加模型信息
        infoDict.merge(modelInfo(currentCls))

        currentCls = parent
    }

    return infoDict
}
