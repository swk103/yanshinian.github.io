代码重构

抽取路径函数

///  返回指定数据的路径
///
///  :param: filename 文件名
///
///  :returns: 文件路径
func dataPath(filename: String) -> String {
    let bundle = NSBundle(forClass: SwiftJSONModelTests.self)
    let path = bundle.pathForResource(filename, ofType: nil)

    XCTAssertNotNil(path, "数据加载错误")
    return path!
}

新建 JSON 函数

  • 在 SwiftJSONModel.swift 中新建 Class->JSONModelManager
  • 新建函数 json
///  将二进制数据反序列化为 NSDictionary 或者 NSArray
///
///  :param: data json 数据
///
///  :returns: 如果数据格式正确，返回 NSDictionary 或者 NSArray，否则返回 nil
class func json(data: NSData) -> AnyObject? {
    return NSJSONSerialization.JSONObjectWithData(data, options: .allZeros, error: nil)
}

重构后的代码如下：

///  测试加载 JSON
func testLoadJSON() {
    let json = NSData(contentsOfFile: dataPath("Model01.json"))
    let model = JSONModelManager.json(json!) as? NSDictionary
    println(model)

    XCTAssertNotNil(model!, "加载模型失败")
    XCTAssert(model!.isKindOfClass(NSDictionary.self), "模型类型不正确")
}

///  测试加载错误的JSON
func testLoadErrorJSON() {
    let json = NSData(contentsOfFile: dataPath("Error.json"))
    let model = JSONModelManager.json(json!) as? NSDictionary

    XCTAssertNil(model, "错误的json解析应该为空")
}
