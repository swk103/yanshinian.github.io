    // 1. 取出模型字典

    obj.setValue(subObj, forKey: k)

} else if type == "NSArray" {
    println("自定义对象数组")
}

注意：在使用 NSClassFromString 需要指定类所在的命名空间，否则会出错！

解决办法

  • 定义命名空间字符串
    /// 模型类的命名空间名称
    var classNamespaceName: String?
  • 转模型之前定义命名空间

修改的测试代码如下：

///  测试字典转模型
func testDict2Model() {
    let manager = JSONModelManager.sharedManager
    manager.classNamespaceName = "SwiftJSONModelTests"

    let json = NSData(contentsOfFile: dataPath("Model01.json"))
    let dict = JSONModelManager.json(json!) as? NSDictionary

    let model = manager.objectWithDictionary(dict!, cls: ModelInfo.self) as! ModelInfo
    XCTAssertNotNil(model, "模型转换失败")

    println(JSONModelManager.sharedManager.objectDictionary(model))
}

自定义数组属性

添加数组属性，测试会发现仍然是通过 KVC 在赋值，达不到字典转模型的目的

解决办法

定义协议

///  字典转模型协议
@objc protocol JSONModelProtocol {
    ///  自定义类映射字典
    ///
    ///  :returns: 映射字典
    static func customClassMapping() -> [String: String]
}

在自定义模型中遵守协议病实现方法

        }
    }
}

修改模型转字典函数

func objectDictionary(obj: AnyObject) -> [String: AnyObject] {
    // 1. 取出模型字典
    let info = fullModelInfo(obj.classForCoder)

    var dict = [String: AnyObject]()
    for (k, v) in info {
        let value: AnyObject? = obj.valueForKey(k)

        if v.isEmpty && value != nil {
            dict[k] = obj.valueForKey(k)
        } else if value != nil {
            let type = NSStringFromClass(value?.classForCoder)
            println("\(k)-\(type)!!!!!!!!!!\(v)")
            if type == "NSArray" || type == "NSMutableArray" {

            } else {
                dict[k] = objectDictionary(value!)
            }
        }
    }

    return dict
}

///  性能测试
func testPerformace() {
    let manager = JSONModelManager.sharedManager
    manager.classNamespaceName = "SwiftJSONModelTests"

    let json = NSData(contentsOfFile: dataPath("Model01.json"))
    let dict = JSONModelManager.json(json!) as? NSDictionary

    println("开始")
    let start = CFAbsoluteTimeGetCurrent()
    for _ in 0...10000 {
        let model = manager.objectWithDictionary(dict!, cls: ModelInfo.self) as! ModelInfo
    }
    let end = CFAbsoluteTimeGetCurrent() - start
    println("耗时 \(end) 秒")
}
